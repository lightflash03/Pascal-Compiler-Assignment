program arrayaverage;
var
  numbers: array[1..5] of integer;
  i, sum : integer;
  average : real;
begin
  write("enter 5 integer values:");
  for i := 1 to 5 do
  begin
    read(numbers[i]);
  end;
  sum := 0;
  for i := 1 to 5 do
  begin
    sum := sum + numbers[2*i];
  end;
  average := sum / 5;
  write("the sum and average are: "); 
  write(sum, average);
end.
